Ruby methods cheat sheet

Arrays Hashes 
           counting items in array:
           #length #count #size

           array = ['one', 'two', 'three']
           array.length => 3
           array.count => 3
           array.size => 3
 
          counting items in hash:
          #length #size

          hash = {"a" => 1, "b" => 2}
          hash.length
          => 2
          hash.size
          => 2
        
           adding items to array:
           #push or << #unshift #insert

           array = [1,2,3]
           array.push(4)
           => [1,2,3,4]
           push & << adds item to end of array
           array.unshift(0)
           => [0,1,2,3,4]
           unshift adds item to beginning of array
           array.insert(4,'four')
           => [1,2,3,'four']
 
          iterating items in hash:
          #each #each_key #each_value

          hash = {"a" => 1, "b" => 2}
          hash.each {|k,v| puts "#{k} is #{v}"}
          => a is 1
             b is 2
          hash.each_key
          => a
             b
          hash.each_value
          => 1
             2
        
          flattening an array:
          #flatten #flatten(level)

          a = [[1,2,3],[4,5,[6,7]]]
          a.flatten
          => [1,2,3,4,5,6,7]
          a.flatten(1)
          => [1,2,3,4,5,[6,7]]
          optional level argument
      
          flattening a hash:
          #flatten #flatten(level)

          hash = {"a" => 1, "b" => [2, "two"]}
          hash.flatten
          => ["a", 1, "b", [2, "two"]]
          hash.flatten(2)
          => ["a", 1, "b", 2, "two"]
          optional level argument
